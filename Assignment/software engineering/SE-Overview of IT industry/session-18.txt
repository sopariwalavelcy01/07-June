Q-1 : Create a file called playlist.txt and write the names of your top 3 favorite songs from Spotify into it using write mode (w).
ANS :
#include <stdio.h>

int main()
{
    FILE *fp;

    fp = fopen("playlist.txt", "w");

    fprintf(fp, "Tum Hi Ho\n");
    fprintf(fp, "Kesariya\n");
    fprintf(fp, "Apna Bana Le\n");

    fclose(fp);

    printf("Songs written to playlist.txt successfully.\n");

    return 0;
}

OUTPUT :

Contents of playlist.txt:
Tum Hi Ho
Kesariya
Apna Bana Le

Q-2 : Open playlist.txt in read mode (r) and display each song name on a separate line in the console.
ANS :
#include <stdio.h>

int main()
{
    FILE *fp;
    char song[100];

    fp = fopen("playlist.txt", "r");

    if(fp == NULL)
    {
        printf("File not found!\n");
        return 0;
    }

    while(fgets(song, sizeof(song), fp) != NULL)
    {
        printf("%s", song);
    }

    fclose(fp);

    return 0;
}

OUTPUT :
Tum Hi Ho
Kesariya
Apna Bana Le


Q-3 : Add two more song names to playlist.txt without deleting the existing ones by opening the file in append mode (a).

Q-4 : Write a program that reads all song names from playlist.txt and prints only those that contain the word 'love' (case-insensitive).<br><br><em><strong>Hint:</strong> Use the 'in' keyword or equivalent string method for filtering.</em>
ANS :
#include <stdio.h>
#include <string.h>
#include <ctype.h>

int main()
{
    FILE *fp;
    char song[100], temp[100];
    int i;

    fp = fopen("playlist.txt", "r");

    if(fp == NULL)
    {
        printf("File not found!\n");
        return 0;
    }

    while(fgets(song, sizeof(song), fp))
    {
        strcpy(temp, song);

        // Convert to lowercase
        for(i = 0; temp[i] != '\0'; i++)
        {
            temp[i] = tolower(temp[i]);
        }

        // Check if "love" exists
        if(strstr(temp, "love") != NULL)
        {
            printf("%s", song);
        }
    }

    fclose(fp);

    return 0;
}

If playlist.txt contains:

Love Story
Tum Hi Ho
Perfect
Love Me Like You Do
Heeriye

OUTPUT  :

Love Story
Love Me Like You Do
